        IDJuego SMALLINT PRIMARY KEY AUTO_INCREMENT,
        Nombre VARCHAR(40) NOT NULL UNIQUE,
        Plataforma SET('PSP','Wii','PlayStation','Xbox','Nintendo DS','PC') COMMENT 'Plataformas soportadas por el juego',
